## install virtualenv
### make sure to have installed
- apt-get install python3-venv -y
- python3

### create virtualenv to current directory
- python3 -m venv .
- this command creates some directories and files, isolating your python machine environment from local 'virtual' environment;
- . ./bin/activate
- activating virtualenv
- (project) user@desktop:~/directory/python$

### add packages using pip
- python3 -m pip install 'package'
